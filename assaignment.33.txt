#include<stdio.h>
void main()
{
	int n,i;
	int a[50];
	printf("enter n value");
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	int len=n-1;
	find(a,0,len);
}
void find(int a[],int start,int end)
{
	if(start>end)
	return;
	else
		printf("%d  ",a[start]);
		find(a,start+1,end);
}
